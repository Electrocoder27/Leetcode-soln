class Solution {
    public int[][] reverseSubmatrix(int[][] grid, int x, int y, int k) {
        int t = x ;
        int b = x +(k-1) ;

        while(t<b){
            for(int j =y;j<y+k;j++) {
                int temp = grid[t][j] ;
                grid[t][j] = grid[b][j] ;
                grid[b][j] = temp ;
            }
            t++ ;
            b-- ;
        }
        return grid ;
    }
}
