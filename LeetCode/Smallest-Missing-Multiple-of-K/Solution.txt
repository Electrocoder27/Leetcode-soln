class Solution {
    public int missingMultiple(int[] nums, int k) {
        int n = nums.length ;
        HashSet<Integer> set = new HashSet<>() ;
        for(int i =0;i<n;i++){
            set.add(nums[i]) ;
        }

        int m = 1 ;
        while(true){
            int multi = k*m ;
            if(!set.contains(multi)){
                return multi ;
            }
            m++;
        }
    }
}
